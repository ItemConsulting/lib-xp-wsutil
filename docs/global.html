expand(exportObject)

Expands the library functionalities to create reusable expansions

Parameters:
Name Type Description
exportObject object

The object that extends the library
Source:
  • wsUtil.js, line 338
Example
// lib/expansion.js
var ws = require('/path/to/wsUtil'); // '/lib/wsUtil'

var expansionObject = { expansion: expansion };
ws.expand(expansionObject);

function expansion() {
  ws.addHandler('open', function(event) { log.info('Expansion says hi!'); });
}
exports.expansion = ws;
// service/expansionService/expansionService.js

var expansionLib = require('expansion').expansion

expansionLib.openWebsockets(exports);
expansionLib.expansion(); // Activate expansion
